Linked Lists

Basic Operations

// Node definition
class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

// Linked List class
class LinkedList {
    Node head;

    // Insert at beginning - O(1)
    void insertFirst(int data) {
        Node newNode = new Node(data);
        newNode.next = head;
        head = newNode;
    }

    // Insert at end - O(n)
    void insertLast(int data) {
        Node newNode = new Node(data);
        if (head == null) {
            head = newNode;
            return;
        }
        Node current = head;
        while (current.next != null) {
            current = current.next;
        }
        current.next = newNode;
    }

    // Delete first node - O(1)
    void deleteFirst() {
        if (head != null) {
            head = head.next;
        }
    }

    // Display list
    void display() {
        Node current = head;
        while (current != null) {
            System.out.print(current.data + " -> ");
            current = current.next;
        }
        System.out.println("null");
    }
}

Stacks & Queues

Stack Implementation

class Stack {
    private ArrayList<Integer> items = new ArrayList<>();

    void push(int x) {
        items.add(x);
    }

    int pop() {
        return items.remove(items.size() - 1);
    }

    int peek() {
        return items.get(items.size() - 1);
    }

    boolean isEmpty() {
        return items.isEmpty();
    }
}

Queue Implementation

// Using Java's built-in Queue
Queue<Integer> queue = new LinkedList<>();

queue.add(10);       // Enqueue
queue.remove();      // Dequeue
queue.peek();        // See front element
queue.isEmpty();     // Check if empty

Trees & Binary Search Trees

Tree Terminology

  • Root: Top node
  • Leaf: Node with no children
  • Height: Distance from node to farthest leaf
  • Balanced: Heights of subtrees differ by at most 1

Binary Search Tree

class Node {
    int val;
    Node left, right;

    Node(int val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

class BST {
    Node root;

    void insert(int val) {
        root = insertHelper(root, val);
    }

    private Node insertHelper(Node node, int val) {
        if (node == null) {
            return new Node(val);
        }

        if (val < node.val) {
            node.left = insertHelper(node.left, val);
        } else {
            node.right = insertHelper(node.right, val);
        }

        return node;
    }
}

Tree Traversals

// Inorder: Left, Root, Right (gives sorted order in BST)
void inorder(Node node) {
    if (node != null) {
        inorder(node.left);
        System.out.print(node.val + " ");
        inorder(node.right);
    }
}

// Preorder: Root, Left, Right
void preorder(Node node) {
    if (node != null) {
        System.out.print(node.val + " ");
        preorder(node.left);
        preorder(node.right);
    }
}

// Postorder: Left, Right, Root
void postorder(Node node) {
    if (node != null) {
        postorder(node.left);
        postorder(node.right);
        System.out.print(node.val + " ");
    }
}

// Level Order (BFS)
void levelOrder(Node root) {
    if (root == null) return;
    Queue<Node> q = new LinkedList<>();
    q.add(root);

    while (!q.isEmpty()) {
        Node node = q.remove();
        System.out.print(node.val + " ");
        if (node.left != null) q.add(node.left);
        if (node.right != null) q.add(node.right);
    }
}

Graphs

Graph Representations

  • Adjacency Matrix: 2D array, good for dense graphs
  • Adjacency List: HashMap/Array of lists, good for sparse graphs

Graph Traversal: DFS & BFS

// Depth-First Search (DFS) - O(V + E)
void dfs(int node, Map<Integer, List<Integer>> graph,
         Set<Integer> visited) {
    visited.add(node);
    System.out.print(node + " ");

    for (int neighbor : graph.getOrDefault(node, new ArrayList<>())) {
        if (!visited.contains(neighbor)) {
            dfs(neighbor, graph, visited);
        }
    }
}

// Breadth-First Search (BFS) - O(V + E)
void bfs(int start, Map<Integer, List<Integer>> graph) {
    Set<Integer> visited = new HashSet<>();
    Queue<Integer> queue = new LinkedList<>();

    queue.add(start);
    visited.add(start);

    while (!queue.isEmpty()) {
        int node = queue.remove();
        System.out.print(node + " ");

        for (int neighbor : graph.getOrDefault(node, new ArrayList<>())) {
            if (!visited.contains(neighbor)) {
                visited.add(neighbor);
                queue.add(neighbor);
            }
        }
    }
}

Dynamic Programming

Top-Down (Memoization)

// Fibonacci with Memoization
Map<Integer, Long> memo = new HashMap<>();

long fib(int n) {
    if (n <= 1) return n;
    if (memo.containsKey(n)) return memo.get(n);

    long result = fib(n-1) + fib(n-2);
    memo.put(n, result);
    return result;
}
// Time: O(n), Space: O(n)

Bottom-Up (Tabulation)

// Fibonacci with Tabulation
long fib(int n) {
    if (n <= 1) return n;

    long[] dp = new long[n + 1];
    dp[0] = 0;
    dp[1] = 1;

    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
    }

    return dp[n];
}
// Time: O(n), Space: O(n)
